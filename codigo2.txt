local activeTimers = {}
local activeCallbacks = {}

local function cancelTimersForFeature(featureName)
    if activeTimers[featureName] then 
        for _, timerId in ipairs(activeTimers[featureName]) do 
            LuaTimer:cancel(timerId) 
        end 
        activeTimers[featureName] = nil 
    end
end

local function cancelCallbacksForFeature(featureName)
    if activeCallbacks[featureName] then 
        for event, callbackId in pairs(activeCallbacks[featureName]) do 
            Listener.unregisterCallBack(event, callbackId) 
        end 
        activeCallbacks[featureName] = nil 
    end
end

local function registerTimer(featureName, timerId)
    if not activeTimers[featureName] then 
        activeTimers[featureName] = {} 
    end 
    table.insert(activeTimers[featureName], timerId)
end

local function registerCallback(featureName, event, callbackId)
    if not activeCallbacks[featureName] then 
        activeCallbacks[featureName] = {} 
    end 
    activeCallbacks[featureName][event] = callbackId
end

local function isPlayerInAir()
    local player = PlayerManager:getClientPlayer()
    if not player or not player.Player then return false end
    
    local pos = player.Player:getPosition()
    local footY = pos.y - 1.7
    
    for i = 0, 1 do
        local checkPos = VectorUtil.newVector3(pos.x, footY - i, pos.z)
        if EngineWorld:getBlockId(checkPos) ~= 0 then
            return false
        end
    end
    
    return true
end

local activeHacks = {}
local hackButtons = {}
local buttonWidth = 150
local buttonHeight = 45
local startX = 20
local startY = 80
local columns = 4
local horizontalSpacing = 10
local verticalSpacing = 10

for i, feature in ipairs(hackFeatures) do
    local col = (i-1) % columns
    local row = math.floor((i-1) / columns)
    local btn = GUIManager:createGUIWindow(GUIType.Button, "HackBtn" .. feature.name)
    btn:SetText("^FF0000" .. feature.name .. ":OFF")
    btn:SetWidth({0, buttonWidth})
    btn:SetHeight({0, buttonHeight})
    btn:SetXPosition({0, startX + col * (buttonWidth + horizontalSpacing)})
    btn:SetYPosition({0, startY + row * (buttonHeight + verticalSpacing)})
    btn:SetBackgroundColor({0, 0, 0, 0.8})
    btn:SetVisible(false)
    btn:SetLevel(9)
    GUIRoot:AddChildWindow(btn)
    activeHacks[feature.name] = false
    hackButtons[feature.name] = btn
    btn:registerEvent(GUIEvent.ButtonClick, function()
        activeHacks[feature.name] = not activeHacks[feature.name]
        if activeHacks[feature.name] then
            btn:SetText("^00FF00" .. feature.name .. ":ON")
            feature.onFunc()
        else
            btn:SetText("^FF0000" .. feature.name .. ":OFF")
            feature.offFunc()
        end
    end)
end

local function unloadAll()
    for featureName, isActive in pairs(activeHacks) do
        if isActive then
            for _, feature in ipairs(hackFeatures) do
                if feature.name == featureName then
                    feature.offFunc()
                    break
                end
            end
        end
    end
    
    if mainButton then
        GUIRoot:RemoveChildWindow(mainButton)
        GUIManager:destroyWindow(mainButton)
    end
    
    for _, btn in pairs(hackButtons) do
        GUIRoot:RemoveChildWindow(btn)
        GUIManager:destroyWindow(btn)
    end
end

return {
    unload = unloadAll
}
