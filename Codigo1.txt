
local pos = VectorUtil.newVector3
function Animate(Gui)
    local scale = 0.5
    Gui:SetScale(pos(scale, scale, scale))    
    local timer = LuaTimer:scheduleTicker(function()
        if scale <= 1 then scale = scale + 0.1 else scale = scale - 0.025 end
        Gui:SetScale(pos(scale, scale, scale))    
    end, 1, 10)
end

function playSound(name, pitch)
    local player = PlayerManager:getClientPlayer()

function toggleHack(btn, hackName, enableFunc, disableFunc, soundOn, soundOff)
    states[hackName] = not states[hackName]
    if states[hackName] then
        btn:SetBackgroundColor({0, 255, 0, 255})
        UIHelper.showToast(green .. hackName .. ": Включено")
        playSound(soundOn or "random.orb", 1.0)
        if enableFunc then enableFunc() end
    else
        btn:SetBackgroundColor({0, 0, 0, 0.8})
        UIHelper.showToast(red .. hackName .. ": Выключено")
        playSound(soundOff or "random.click", 0.8)
        if disableFunc then disableFunc() end
        if activeTimers[hackName] then
            LuaTimer:cancel(activeTimers[hackName])
            activeTimers[hackName] = nil
        end
    end
end

local hue = 0
local function interpolateColor(hue)
    local r, g, b = 0, 0, 0
    if hue < 60 then r, g, b = 1, hue / 60, 0
    elseif hue < 120 then r, g, b = (120 - hue) / 60, 1, 0
    elseif hue < 180 then r, g, b = 0, 1, (hue - 120) / 60
    elseif hue < 240 then r, g, b = 0, (240 - hue) / 60, 1
    elseif hue < 300 then r, g, b = (hue - 240) / 60, 0, 1
    else r, g, b = 1, 0, (360 - hue) / 60 end
    return r, g, b
end

        local r, g, b = interpolateColor(hue)
        updateTitle:SetTextColor({r, g, b, 2.7})
    end
    LuaTimer:scheduleTimer(UpdateColor, 50, -1)
end

local function addItem(name, index, func, scale)
    local itemRow = 6; local horizontalSpacing = 160; local verticalSpacing = 50
    local startX = 250; local startY = 100
    local row = math.floor((index - 1) / itemRow); local col = (index - 1) % itemRow
    local positionX = startX + (col * horizontalSpacing); local positionY = startY + (row * verticalSpacing)
    local Button = GUIManager:createGUIWindow(GUIType.StaticText, tostring(math.random(1, 999e9)))
    Button:SetHorizontalAlignment(HorizontalAlignment.Left); Button:SetVerticalAlignment(VerticalAlignment.Top)
    Button:SetTextHorzAlign(HorizontalAlignment.Center); Button:SetTextVertAlign(VerticalAlignment.Center)
    Button:SetHeight({0, 40}); Button:SetWidth({0, 155})
    Button:SetYPosition({0, positionY}); Button:SetXPosition({0, positionX})
    Button:registerEvent(GUIEvent.Click, func)
    GUISystem.Instance():GetRootWindow():AddChildWindow(Button)
    Button:SetLevel(0); Button:SetBordered(true); Button:SetText(name); Button:SetTouchable(true); Button:SetVisible(false)
    Button:SetTextScale(scale or 1); Button:SetBackgroundColor({0, 0, 0, 0.8})

    local hue = 0

        Button:SetTextColor({r, g, b, 0.6})
    end
    LuaTimer:scheduleTimer(UpdateColor, 50, -1)
    return Button
end

function Gmpanel.saveTP()
    local player = PlayerManager:getClientPlayer()
    if player and player.Player then
        savedPositions.saveTP = player.Player:getPosition()
        UIHelper.showToast("^00FF00Позиция сохранена! (F1)")
        playSound("random.orb", 1.0)
    end
end
function Gmpanel.backTP()
    if savedPositions.saveTP then
        local player = PlayerManager:getClientPlayer()
        if player and player.Player then
            player.Player:setPosition(savedPositions.saveTP)
            UIHelper.showToast("^00FF00Телепорт назад! (F2)")
            playSound("random.orb", 1.0)
        end
    else
        UIHelper.showToast("^FF0000Нет сохранённой позиции!")
        playSound("random.click", 0.8)
    end
end


GMHelper.boostFPSActive = false
function Gmpanel.enableBoostFPS()
   
    ClientHelper.putIntPrefs("GraphicsQuality", 0)
    ClientHelper.putBoolPrefs("EnableParticles", false)
    ClientHelper.putBoolPrefs("EnableWeather", false)
    ClientHelper.putBoolPrefs("EnableShadows", false)
    ClientHelper.putIntPrefs("RenderDistance", 4)
    ClientHelper.putBoolPrefs("SmoothLighting", false)
    ClientHelper.putBoolPrefs("FancyGraphics", false)
    
    
    for blockId = 1, 1000 do
        local block = BlockManager.getBlockById(blockId)
        if block then
            block:setLightValue(0, 0, 0)
        end
    end
    
    UIHelper.showToast("^00FF00Boost FPS: Включено")
end

function Gmpanel.disableBoostFPS()
    -- Восстановление стандартных настроек
    ClientHelper.putIntPrefs("GraphicsQuality", 2)
    ClientHelper.putBoolPrefs("EnableParticles", true)
    ClientHelper.putBoolPrefs("EnableWeather", true)

    ClientHelper.putBoolPrefs("SmoothLighting", true)
    ClientHelper.putBoolPrefs("FancyGraphics", true)
    
    UIHelper.showToast("^FF0000Boost FPS: Выключено")
end

-- Cannon
GMHelper.CannonBtn = function(self, text)
    local canon = GUIManager:getWindowByName("Main-Cannon")
    self.cannonActive = not (self.cannonActive or false)
    
    if text then
        text:SetBackgroundColor(self.cannonActive and {0, 0, 255, 255} or {0, 0, 0, 0.8})
    end
    
    if self.cannonActive then
        canon:SetVisible(true)
        local scale = 0.5
        canon:SetScale(VectorUtil.newVector3(scale, scale, scale))
        
        self.scaleTimer = LuaTimer:scheduleTicker(function()
            if scale <= 1 then scale = scale + 0.05 else scale = scale - 0.0125 end
            canon:SetScale(VectorUtil.newVector3(scale, scale, scale))
        end, 1, 15)
        
        canon:registerEvent(GUIEvent.ButtonClick, function()
            local clientPlayer = PlayerManager:getClientPlayer()
            if clientPlayer then
                local pitch = clientPlayer.Player:getPitch()
                local yaw = clientPlayer.Player:getYaw()
                local pitchRad = pitch * math.pi / 180
                local yawRad = yaw * -math.pi / 180
                local x = math.cos(pitchRad) * math.sin(yawRad) * 3
                local y = -math.sin(pitchRad) * 4
                local z = math.cos(pitchRad) * math.cos(yawRad) * 3
                local newPos = VectorUtil.newVector3(x, y, z)
                clientPlayer.Player:setVelocity(newPos)
                SoundUtil.playSound(313)
            end
        end)
        UIHelper.showToast("^00FF00Cannon: Enabled")
    else
        canon:SetVisible(false)
        if self.scaleTimer then
            LuaTimer:cancel(self.scaleTimer)
            self.scaleTimer = nil
        end
        UIHelper.showToast("^FF0000Cannon: Disabled")
    end
end

            local player = PlayerManager:getClientPlayer()
            if not player or not player.Player then return end
            local yaw = player.Player:getYaw()
            local pitch = player.Player:getPitch()
            local yawRad = math.rad(yaw)
            local pitchRad = math.rad(pitch)
            local speed = 2.1
            local x = -speed * math.cos(pitchRad) * math.sin(yawRad)
            local y = -speed * math.sin(pitchRad)
            local z = speed * math.cos(pitchRad) * math.cos(yawRad)
            player.Player:setVelocity(VectorUtil.newVector3(x, y, z))
        end, 15, -1)

    else
        menuBtn:SetBackgroundColor({0, 0, 0, 0.8})
        UIHelper.showToast("^FF0000JetPack: Выключен")
        playSound("random.click", 0.8)

        if self.jetpackTimer then
            LuaTimer:cancel(self.jetpackTimer)
            self.jetpackTimer = nil
        end

        if self.jetpackButton then
            self.jetpackButton:SetVisible(false)
        end
    end
end

local function addTab(name, index, TabFunction)
    local verticalSpacing = 51; local startY = 100
    local positionY = startY + ((index - 1) * verticalSpacing)
    local Button = GUIManager:createGUIWindow(GUIType.StaticText, tostring(math.random(1, 999e9)))
    Button:SetHorizontalAlignment(HorizontalAlignment.Left); Button:SetVerticalAlignment(VerticalAlignment.Top)
    Button:SetTextHorzAlign(HorizontalAlignment.Center); Button:SetTextVertAlign(VerticalAlignment.Center)
    Button:SetHeight({0, 46}); Button:SetWidth({0, 159})
    Button:SetYPosition({0, positionY}); Button:SetXPosition({0, 45})
    Button:registerEvent(GUIEvent.Click, TabFunction)
    GUISystem.Instance():GetRootWindow():AddChildWindow(Button)
    Button:SetLevel(0); Button:SetBordered(true); Button:SetText(name); Button:SetTouchable(true); Button:SetVisible(false)
    Button:SetBackgroundColor({0, 0, 0, 0.8})

    local hue = 0
    local function UpdateColor()
        hue = (hue + 0.5) % 360
        local r, g, b = interpolateColor(hue)
        Button:SetTextColor({r, g, b, 0.6})
    end
    LuaTimer:scheduleTimer(UpdateColor, 10, -1)
    return Button
end

function Gmpanel.enableAimBot()
    activeTimers["Aimbot"] = LuaTimer:scheduleTimer(function()
        local me = PlayerManager:getClientPlayer()
        if not me then return end
        local myPos = me.Player:getPosition()
        local myTeamId = me.Player:getTeamId()
        local closestPlayer, closestDistance = nil, math.huge
        for _, player in pairs(PlayerManager:getPlayers()) do
            if player ~= me and player.Player and player.Player:getTeamId() ~= myTeamId then
                local dist = MathUtil:distanceSquare2d(player:getPosition(), myPos)
                if dist < closestDistance then
                    closestDistance = dist
                    closestPlayer = player
                end
            end
        end
        if closestPlayer and closestDistance < 60 then
            local camera = SceneManager.Instance():getMainCamera()
            local pos = camera:getPosition()
            local dir = VectorUtil.sub3(closestPlayer:getPosition(), pos)
            local yaw = math.atan2(dir.x, dir.z) / math.pi * -180
            local calculate = math.sqrt(dir.x * dir.x + dir.z * dir.z)
            local pitch = -math.atan2(dir.y + 1.5, calculate) / math.pi * 180
            me.Player.rotationYaw = yaw
            me.Player.rotationPitch = pitch
            CGame.Instance():handleTouchClick(800, 360)
        end
    end, 50, -1)
end
function Gmpanel.disableAimBot() end

function Gmpanel.enableAntiKnockback()
    states.lastHealth = nil
    states.lastPosition = nil
    activeTimers["AntiKnockBack"] = LuaTimer:scheduleTimer(function()
        local me = PlayerManager:getClientPlayer()
        if not me or not me.Player then return end
        local currentHealth = me.Player:getHealth()
        local currentPosition = me.Player:getPosition()
        if states.lastHealth and currentHealth < states.lastHealth and states.lastPosition then
            me.Player:setPosition(states.lastPosition)
        end
        states.lastHealth = currentHealth
        states.lastPosition = currentPosition
    end, 10, -1)
end
function Gmpanel.disableAntiKnockback()
    states.lastHealth = nil
    states.lastPosition = nil
end

function Gmpanel.enableAntiVoid()
    states.antiVoidPos = nil
    activeTimers["AntiVoid"] = LuaTimer:scheduleTimer(function()
        local player = PlayerManager:getClientPlayer().Player
        if player.onGround then
            states.antiVoidPos = player:getPosition()
        elseif states.antiVoidPos and player:getPosition().y <= 1 then
            player.noClip = true
            player:setPosition(VectorUtil.newVector3(states.antiVoidPos.x, states.antiVoidPos.y + 2, states.antiVoidPos.z))
            LuaTimer:schedule(function() player.noClip = false end, 100)
        end
    end, 10, -1)
end
function Gmpanel.disableAntiVoid()
    states.antiVoidPos = nil
